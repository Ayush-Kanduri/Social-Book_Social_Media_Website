module.exports.home = (request, response) => {
	//Populate the User (from user field) of each Post.
	Post.find({})
		.populate("user")
		.exec((err, posts) => {
			if (err) {
				console.log("Error in Fetching Posts & User");
				return;
			}
			return response.render("home", {
				title: "Home Page",
				posts: posts,
			});
		});
};
